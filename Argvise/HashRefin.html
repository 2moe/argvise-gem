require 'argvise'
class A
  using Argvise::HashRefin
  def self.demo
    puts({ target: "wasm32-wasip2" }.to_argv)
      # => ["--target", "wasm32-wasip2"]

    puts({ target: "wasm32-wasip2" }.to_argv_bsd)
      # => ["-target", "wasm32-wasip2"]

    puts({}.respond_to?(:to_argv)) #=> true
  end
end
